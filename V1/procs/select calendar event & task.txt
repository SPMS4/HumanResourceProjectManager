CREATE DEFINER=`root`@`localhost` PROCEDURE `SelectTask`(IN `exGroupID` INT, IN `exGroupID` INT, OUT `iEndDate` INT)
    NO SQL
BEGIN

Select iEndDate = EndDate 
From taskevents
Where TaskEventID = exTaskEventID;

Select *
From TaskEvents
Where EndDate = iEndDate and GroupID = exGroupID;

END




CREATE DEFINER=`root`@`localhost` PROCEDURE `SelectEvent`()
    NO SQL
BEGIN

/*Select iEndDate = EndDate

From TaskEvents
Where EndDate = null;

END
